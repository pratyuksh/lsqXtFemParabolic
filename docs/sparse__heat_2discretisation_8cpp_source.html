485 void sparseHeat::LsqSparseXtFem
486 :: assembleSourceWithTemporalGradientOfTemperatureBasis(Vector& b) const
487 {
488  auto spatialFes = m_spatialNestedFEHierarchyTemperature->getFESpaces();
489 
490  // auxiliary variables
491  Vector buf;
492  int shift = 0;
493 
494  // temporal integration rule
495  int order = 4;
496  const IntegrationRule *ir
497  = &IntRules.Get(Geometry::SEGMENT, order);
498 
499  // coarsest temporal + finest spatial level
500  // standard FE basis at the coarsest temporal level
501  {
502  int numTemporalMeshElements
503  = static_cast<int>(std::pow(2, m_minTemporalLevel));
504  double ht = m_endTime / numTemporalMeshElements;
505 
506  int spatialId = getSpatialIndex(0);
507  auto curSpatialFes = spatialFes[spatialId];
508  int curSpatialFesDim = curSpatialFes->GetTrueVSize();
509 
510  buf.SetSize(curSpatialFesDim);
511  for (int n=0; n<numTemporalMeshElements; n++)
512  {
513  double tLeft = n*ht;
514  double tRight = (n+1)*ht;
515 
516  Vector tmp1(b.GetData() + n*curSpatialFesDim + shift,
517  curSpatialFesDim);
518  Vector tmp2(b.GetData() + (n+1)*curSpatialFesDim + shift,
519  curSpatialFesDim);
520 
521  for (int i=0; i<ir->GetNPoints(); i++)
522  {
523  const IntegrationPoint &ip = ir->IntPoint(i);
524  double t = affineTransform(tLeft, tRight, ip.x);
525 
526  assembleSourceWithTemporalGradientOfTemperatureBasisAtGivenTime
527  (t, curSpatialFes, buf);
528 
529  // tmp1 += weight * temporalGradient1 * ht * ft
530  // temporalGradient1 * ht = -1
531  tmp1.Add(-ip.weight, buf);
532 
533  // tmp2 += weight * temporalGradient2 * ht * ft
534  // temporalGradient2 * ht = +1
535  tmp2.Add(+ip.weight, buf);
536 
537 // std::cout << n << "\t"
538 // << tLeft << "\t"
539 // << tRight << "\t"
540 // << i << "\t"
541 // << t << std::endl;
542  }
543  }
544  shift += (numTemporalMeshElements+1)*curSpatialFesDim;
545  }
546 
547  // hierarchical temporal basis for the remaining levels
548  for (int m=1; m<m_numLevels; m++)
549  {
550  int numTemporalMeshElements
551  = static_cast<int>(std::pow(2, m_minTemporalLevel + m));
552  double ht = m_endTime / numTemporalMeshElements;
553 
554  int spatialId = getSpatialIndex(m);
555  auto curSpatialFes = spatialFes[spatialId];
556  int curSpatialFesDim = curSpatialFes->GetTrueVSize();
557 
558  buf.SetSize(curSpatialFesDim);
559  for (int n=0; n<numTemporalMeshElements; n++)
560  {
561  double tLeft = n*ht;
562  double tRight = (n+1)*ht;
563 
564  int nn = n/2;
565  Vector tmp(b.GetData() + nn*curSpatialFesDim + shift, curSpatialFesDim);
566 
567  for (int i=0; i<ir->GetNPoints(); i++)
568  {
569  const IntegrationPoint &ip = ir->IntPoint(i);
570  double t = affineTransform(tLeft, tRight, ip.x);
571 
572  assembleSourceWithTemporalGradientOfTemperatureBasisAtGivenTime
573  (t, curSpatialFes, buf);
574 
575  if (n%2 == 0) {
576  // tmp += weight * temporalGradient * ht * f(t)
577  // temporalGradient * h = +1
578  tmp.Add(+ip.weight, buf);
579  }
580  else {
581  // tmp += weight * temporalGradient * ht * ft
582  // temporalGradient * h = -1
583  tmp.Add(-ip.weight, buf);
584  }
585  }
586  }
587  shift += (numTemporalMeshElements/2)*curSpatialFesDim;
588  }
