281 void sparseHeat::LsqSparseXtFem
282 :: buildSystemBlock11()
283 {
284  auto temporalBlockSizes
285  = evalTemporalBlockSizes(m_minTemporalLevel,
286  m_maxTemporalLevel);
287  auto spatialBlockSizes
288  = m_spatialNestedFEHierarchyTemperature->getNumDims();
289 
290  auto blockSizes = evalSpaceTimeBlockSizes(spatialBlockSizes,
291  temporalBlockSizes);
292  auto blockOffsets = evalBlockOffsets(blockSizes);
293 
294  auto buf = std::make_unique<BlockMatrix>(blockOffsets);
295  // block11 = M^t_{\ell1, \ell2} \otimes K^{x;(1,1)}_{L-\ell1, L-\ell2}
296  for (int i=0; i<m_numLevels; i++)
297  {
298  int ii = getSpatialIndex(i);
299  for (int j=0; j<m_numLevels; j++)
300  {
301  int jj = getSpatialIndex(j);
302  auto mat = OuterProduct(m_temporalMass->GetBlock(i, j),
303  m_spatialStiffness1->GetBlock(ii, jj));
304  buf->SetBlock(i, j, mat);
305  }
306  }
307  buf->owns_blocks = true;
308  m_systemBlock11 = buf->CreateMonolithic();
309 
310  // block11 += (K^t_{\ell1, \ell2} + E^t_{\ell1, \ell2})
311  // \otimes M^{x; (1,1)}_{L-\ell1, L-\ell2}
312  buf.reset(new BlockMatrix(blockOffsets));
313  for (int i=0; i<m_numLevels; i++)
314  {
315  int ii = getSpatialIndex(i);
316  for (int j=0; j<m_numLevels; j++)
317  {
318  int jj = getSpatialIndex(j);
319  auto tmp = Add(m_temporalInitial->GetBlock(i, j),
320  m_temporalStiffness->GetBlock(i, j));
321  auto mat = OuterProduct(*tmp, m_spatialMass1->GetBlock(ii, jj));
322  buf->SetBlock(i, j, mat);
323  delete tmp;
324  }
325  }
326  buf->owns_blocks = true;
327  auto tmp = buf->CreateMonolithic();
328  m_systemBlock11->Add(1., *tmp);
329  delete tmp;
